--유저 테이블 생성
CREATE TABLE stuser (
stuser_no NUMBER(38) NOT NULL, 
stuser_id VARCHAR2(500) NOT NULL, 
stuser_pw VARCHAR2(500) NOT NULL, 
stuser_nick VARCHAR2(500) NOT NULL, 
stuser_gender VARCHAR2(500) NOT NULL, 
stuser_age NUMBER(38) NOT NULL, 
stuser_loc VARCHAR2(500) NOT NULL, 
stuser_date DATE NOT NULL);

--유저 번호 기본키 지정
ALTER TABLE stuser ADD CONSTRAINT stuser_no_pk PRIMARY KEY (stuser_no);

--유저 번호 시퀀스 생성
CREATE SEQUENCE seq_stuser_no;

--게시글 테이블 생성
CREATE TABLE stboard (
	stboard_idx NUMBER(38) NOT NULL, 
	stboard_title VARCHAR2(200) NOT NULL, 
	stboard_content VARCHAR2(4000) NOT NULL, 
	stboard_hit NUMBER(38) NOT NULL, 
	stboard_category VARCHAR2(500) NOT NULL, 
	stboard_recommend NUMBER(38) NOT NULL, 
	stboard_report NUMBER(38) NOT NULL, 
	stboard_date DATE NOT NULL,
	stboardname_no NUMBER(38) NOT NULL,
	stuser_no NUMBER(38) NOT NULL
);

--게시글 번호 기본키 지정
ALTER TABLE stboard ADD CONSTRAINT stboard_idx_pk PRIMARY KEY (stboard_idx);

--게시글 테이블에서 참조할 유저 테이블의 유저 번호 외래키 지정
ALTER TABLE stboard ADD CONSTRAINT stuser_no_fk FOREIGN KEY (stuser_no) REFERENCES stuser(stuser_no);

--게시글 번호 시퀀스 생성
CREATE SEQUENCE seq_stboard_idx;

--게시판 테이블 생성
CREATE TABLE stboardname (
	stboardname_no NUMBER(38) NOT NULL,
	stboardname_name VARCHAR2(60) NOT NULL
);

--게시판 번호 기본키 지정
ALTER TABLE stboardname ADD CONSTRAINT stboardname_no_pk PRIMARY KEY (stboardname_no);

--게시글 테이블에서 참조할 게시판 테이블의 게시판 번호 외래키 지정
ALTER TABLE stboard ADD CONSTRAINT stboardname_no_fk FOREIGN KEY (stboardname_no) REFERENCES stboardname(stboardname_no);
